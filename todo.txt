derivadas:
(+a)' = a'
(a+b)' = a'+b'
(-a)' = -a'
(a-b)' = a'-b'
(a%)' = a'%
(a*b)' = a'*b + a*b'
(a/b)' = (a'*b-a*b')/b²
(a^b)' = a^(b-1)*(a'*b+a*b'*ln(a))
(a²)' = 2*a*a'
(a³)' = 3*a^2*a'
(a\xb)' = a'\xb + a\xb'
(\v/a)' = a'/(2*\v/a)
acos'(x) = -x'/sqrt(1-x²)
acosh'(x) = x'/sqrt((x-1)*(x+1))
asin'(x) = x'/sqrt(1-x²)
asinh'(x) = x'/sqrt(x²+1)
atan'(x) = x'/(x²+1)
atan2'(a, b) = (a'*b-a*b')/(a²+b²)
atanh'(x) = x'/(1-x²)
cis'(x) = x'*i*cis(x)
cos'(x) = -x'*sin(x)
cosh'(x) = x'*sinh(x)
exp'(x) = x'*exp(x)
ln'(x) = x'/x
log'(a, b) = (a'/a*ln(b)-b'/b*ln(a))/ln(b)²
log10'(x) = x'/(x*ln(10))
pow'(a, b) = a^(b-1)*(a'*b+a*b'*ln(a))
q'(x) = x'
sin'(x) = x'*cos(x)
sinh'(x) = x'*cosh(x)
sqrt'(x) = x'/(2*sqrt(x))
tan'(x) = x'/cos(x)²
tanh'(x) = x'/cosh(x)²
if'(oq, casoSim, casoNao) = if(oq, casoSim', casoNao')
cross'(a, b) = cross(a', b)+cross(a, b')
dot'(a, b) = dot(a', b)+dot(a, b')
